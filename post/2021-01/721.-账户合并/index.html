HashMap去重加检索，并查集合并用户，而且题目要求返回邮箱按字符串大小顺序排列，所以又加了TreeSet去重加排序，另外第一次尝试用java的迭代器iterator进行遍历操作，整个写下来感觉对java集合方面提升很大。

class Solution {
    int[] parent;
    
    public List<List<String>> accountsMerge(List<List<String>> accounts) {
        // 邮箱指向用户的哈希表，和并查集配合，确定哪些账户需要合并
        HashMap<String, Integer> map = new HashMap<>();
        // 用户指向邮箱的哈希表，第二次遍历时用来存储每个用户的邮箱，并完成对邮箱的排序
        HashMap<Integer, TreeSet<String>> emails = new HashMap<>();
        List<List<String>> res = new ArrayList<>();
        parent = new int[accounts.size()];      // 并查集
        for(int i = 0;i<accounts.size();i++) parent[i] = i;
        Iterator<List<String>> iterators = accounts.iterator();
        int i = 0;      // 记录遍历时的当前用户
        while(iterators.hasNext()) {
            Iterator<String> iterator = iterators.next().iterator();
            iterator.next();        // accounts[i][0]是用户名，不需要遍历
            while(iterator.hasNext()) {
                String email = iterator.next();
                if(map.containsKey(email)) union(i, map.get(email));    // 若该邮箱已在哈希表中，则合并两个账户
                else map.put(email, i);                                 // 否则将邮箱放入哈希表
            }
            i++;
        }
        Iterator<Map.Entry<String, Integer>> entryIterator = map.entrySet().iterator();
        while(entryIterator.hasNext()) {
            Map.Entry<String, Integer> entry = entryIterator.next();
            // 若该用户已在表中，则将邮箱加入该用户的邮箱集合中
            if(emails.containsKey(find(entry.getValue()))) emails.get(find(entry.getValue())).add(entry.getKey());
            // 否则新建用户，并将该邮箱作为该用户邮箱集合的初始值
            else emails.put(find(entry.getValue()), new TreeSet<>(Arrays.asList(entry.getKey())));
        }
        Iterator<Map.Entry<Integer, TreeSet<String>>> emailsIterator = emails.entrySet().iterator();
        while(emailsIterator.hasNext()) {       //生成返回值
            Map.Entry<Integer, TreeSet<String>> entry = emailsIterator.next();
            List<String> temp = new ArrayList<>();
            temp.add(accounts.get(entry.getKey()).get(0));
            temp.addAll(entry.getValue());
            res.add(temp);
        }
        return res;
    }

    public int find(int i) {
        if(parent[i]!=i) parent[i] = find(parent[i]);
        return parent[i];
    }

    public void union(int i1, int i2) {
        parent[find(i1)] = find(i2);
    }
}

执行用时超过91%，内存消耗超过84%，还是比较满意的。

另外，对集合进行排序，除了TreeSet（SortedSet的唯一实现类，红黑树实现），还有Collections.sort()（对列表排序），其他还有不少没接触过的类和接口。
