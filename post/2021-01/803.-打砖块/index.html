class Solution:
    def hitBricks(self, grid: List[List[int]], hits: List[List[int]]) -> List[int]:
        # 逆向思考，先把要消除的砖块先全部打掉，然后逆序补上要消除的砖块，
        # 根据补上的砖块进行遍历，计算掉落砖块的数目。
        # grid数组中，被打掉、不存在的砖块为0，存在但尚未遍历的砖块为1，已遍历的砖块为2
        
        drop = []
        current = 0
        for hit in hits:                            # 先把将要消除的砖块全部打掉
            grid[hit[0]][hit[1]] = grid[hit[0]][hit[1]] - 1
        for j in range(len(grid[0])):               # 遍历最终依然保持稳定的砖块
            visit(0, j, grid)
        for i in range(len(hits)-1, -1, -1):        # 逆序补上要消除的砖块
            x, y = hits[i][0], hits[i][1]
            grid[x][y] = grid[x][y] + 1
            if not grid[x][y]:                      # 补上后仍为0，说明该砖块原本就不存在
                drop.append(0)
                continue
            if x!=0 and not judge(x-1, y, grid) + judge(x+1, y, grid) + \
                    judge(x, y-1, grid) + judge(x, y+1, grid):  # 如果该砖块没有连接顶部且周围不存在已被遍历过的砖块（稳定的砖块），则说明该砖块在之前消除某个砖块时已掉落
                drop.append(0)
                continue
            current = visit(x, y, grid)             # 遍历该砖块连通的尚未遍历过的砖块
            drop.append(current - 1)                # 掉落数为遍历的砖块个数减去自己本身
        drop.reverse()
        return drop


def visit(i, j, grid) -> int:                       # dfs遍历砖块周围尚未遍历过的砖块
    if i<0 or i>=len(grid) or j<0 or j>=len(grid[0]) or grid[i][j]!=1:
        return 0
    grid[i][j] = grid[i][j] + 1
    return visit(i-1, j, grid) + visit(i+1, j, grid) + \
            visit(i, j-1, grid) + visit(i, j+1, grid) + 1


def judge(i, j, grid) -> int:                       # 判断该砖块是否已遍历（稳定）
    if 0<=i<len(grid) and 0<=j<len(grid[0]) and grid[i][j]==2:
        return 1
    return 0
